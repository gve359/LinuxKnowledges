awk - язык обработки строк по шаблонам. Обычно для таблиц


ПРИМЕРЫ

вывести 1й столбец из таблицы
awk '{print $1 }' 

суммирование списка чисел
awk '{s+=$1} END {print s}'

подсчёт количества значений в таблице
awk '$2=="4712" {cnt++} END {print cnt}' awktst.txt
